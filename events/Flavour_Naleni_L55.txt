country_event = {
	id = naleni.11
	title = naleni.11.t
	desc = naleni.11.d
	picture = MILITARY_CAMP_eventPicture
	fire_only_once = yes
	is_triggered_only = yes
	immediate = {
		set_country_flag = L55_riva_reform
		hidden_effect = {
			if = {
			limit = {
				NOT = { check_variable = { which = matauth value = 4 } }
			}
			set_country_flag = L55_low_authority
		}
		if = {
			limit = {
				check_variable = { which = matauth value = 4 }
				NOT = { check_variable = { which = matauth value = 8 } }
			}
			set_country_flag = L55_med_authority
		}
		if = {
			limit = {
				check_variable = { which = matauth value = 8 }
			}
			set_country_flag = L55_high_authority
		}
		set_variable = { which = rebelamount value = 3 }
		}
	}
	option = {
		name = naleni.11.a
		ai_chance = { factor = 100 }
		custom_tooltip = mutiny_event_tt
		hidden_effect = {
			change_government = monarchy
			add_government_reform = naleni_thalassocracy_reform
			add_stability = -2
			if = {
				limit = { has_country_flag = low_authority }
				define_ruler = {
					name = "Ríva"
					dynasty = "Nephiuéda"
					adm = 5
					dip = 4
					mil = 6
					age = 22
					claim = 40
					culture = naleni_harpy
					female = yes
				}
				add_ruler_personality = inspiring_leader_personality
				while = {
					limit = {
						check_variable = { rebelamount = 0 }
					}
					random_core_province = {
					spawn_rebels = {
						size = 3
						type = particularist_rebels
						}
					}
					change_variable = { which = rebelamount value = -1 }
				}
			}
			else_if = {
				limit = { has_country_flag = med_authority }
				define_ruler = {
					name = "Ríva"
					dynasty = "Nephiuéda"
					adm = 5
					dip = 4
					mil = 6
					age = 22
					claim = 50
					culture = naleni_harpy
					female = yes
				}
				add_ruler_personality = inspiring_leader_personality
				while = {
					limit = {
						check_variable = { rebelamount = 0 }
					}
					random_core_province = {
					spawn_rebels = {
						size = 3
						type = particularist_rebels
						}
					}
					change_variable = { which = rebelamount value = -1 }
				}
			}
			else = { 
				define_ruler = {
					name = "Ríva"
					dynasty = "Nephiuéda"
					adm = 5
					dip = 4
					mil = 6
					age = 22
					claim = 60
					culture = naleni_harpy
					female = yes
				}
				add_ruler_personality = inspiring_leader_personality
				while = {
					limit = {
						check_variable = { rebelamount = 0 }
					}
					random_core_province = {
