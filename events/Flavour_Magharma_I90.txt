#Reign of Ziltagh
country_event = {
	id = flavor_maghargma.1
	title = flavor_maghargma.1.t
	picture = DIPLOMACY_SIGNING_eventPicture

	is_triggered_only = yes

		ai_chance = { factor = 100 }

			duration = -1
		}
	}
}
