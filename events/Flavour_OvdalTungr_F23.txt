# PTM manager event, meant to avoid having MTTH events for OT.
province_event = {
	id = flavour_ovdal_tungr.1001
	title = flavour_ovdal_tungr.1001.t
	desc = flavour_ovdal_tungr.1001.d
	picture = BATTLE_eventPicture
	
	is_triggered_only = yes

	hidden = yes

	immediate = {
		owner = {
			if = { limit = { tag = F23 }
				if = { limit = { NOT = { has_country_flag = F23_grun_strongbellow_arrived } }
					export_to_variable = {
						variable_name = OT_grun_arrival
						value = trigger_value:is_year
					}
					log = "OT year check: [Root.OT_grun_arrival.GetValue]"
					export_to_variable = { 
						variable_name = OT_current_innovativeness
						value = trigger_value:innovativeness
					}
					log = "OT inno: [Root.OT_current_innovativeness.GetValue]"
					multiply_variable = { # Multiplying OT_grun_arrival by 2 instead of dividing current inno by 2, since division is more computationally intensive.
						which = OT_grun_arrival
						value = 2
					}
					change_variable = {
						which = OT_grun_arrival
						which = OT_current_innovativeness
					}
					if = { limit = { check_variable = { which = OT_grun_arrival value = 3000 } }
						country_event = { id = flavour_ovdal_tungr.7 days = 0 random = 360 }
					}
				}
				else = {
					if = { limit = { is_year = 1589 NOT = { has_country_flag = F23_grun_death_delayed } }
						export_to_variable = {
							variable_name = OT_year_check
							value = trigger_value:is_year
						}

						country_event = { id = flavour_ovdal_tungr.10 days = 0 random = 0 }
					}
				}
				if = { limit = { NOT = { has_country_flag = F23_grun_strongbellow_dead } } # No more "MTTH" events after Grun is dead.
					526 = { add_province_triggered_modifier = F23_MTTH_events_manager }
				}
			}
			else = {
				526 = { add_province_triggered_modifier = F23_MTTH_events_manager }
			}
		}
	}
	
	option = { 
		name = flavour_ovdal_tungr.1001.a
		ai_chance = { factor = 100 }
	}
}
