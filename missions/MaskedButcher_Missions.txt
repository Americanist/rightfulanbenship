		icon = mission_burning_man

			custom_tooltip = mb_d_seek

					name = mb_plunderbonus
					duration = 9125
				}
			}
			else_if = {
				limit = {
					has_reform = tribal_assimilation_reform
				}
				add_country_modifier = {
					name = mb_assimilationbonus
					duration = 9125
				}
			}
			else_if = {
				limit = {
					has_reform = learning_by_doing_reform
				}
				add_country_modifier = {
					name = mb_learningbonus
					duration = 9125
				}
			}
			else_if = {
				limit = {
					has_reform = rein_in_the_tribes_reform
				}
				add_country_modifier = {
					name = mb_reinbonus
					duration = 9125
				}
			}
			else = {
				custom_tooltip = mb_d_seek_nochoice
			}
			4033 = {
				set_province_flag = mb_vampire_expedition_flag
				spawn_expedition = yes
			}
		}
	}
