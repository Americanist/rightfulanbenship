		icon = mission_harimar
 		position = 1
 		required_missions = {
		}

		trigger = {
			OR = {
				AND ={ 
					overlord = {
						has_opinion = {
							who = ROOT
							value = 200
						}
						any_subject_country = {
							is_subject_of_type = great_daimyo_vassal
							alliance_with = ROOT
						}
							
					}
					has_country_modifier = raj_mandate_right_to_expansion
				}
				AND = {
					is_vassal = no
					num_of_allies = 2
					4362 = {
						has_spy_network_in = {
							who = owner
							value = 20
						}
					}				
				}
			}
		}

		effect = {
			4362 = {
				add_permanent_claim = ROOT
			}
			4361 = {
				add_permanent_claim = ROOT
			}
			4364 = {
				add_permanent_claim = ROOT
			}
			if = {
				limit = { 
					is_vassal = yes
				}
				add_favors = {
					who = overlord
					amount = 10
				}
			}
			if = {
				limit = {
					is_vassal = no
				}
				every_ally = {
					add_favors = {
						who = ROOT
						amount = 10
					}
				}
			}
		}
	}

	pasiragha_develop_port = {
		icon = mission_harimar
		position = 3

			pasiragha_coast_war
		}
