rahen_religious_turmoil = {
	potential = {
		religion_group = raheni
		has_global_flag = suhan_goes_sicko_mode
		NOT = {
			OR = {
				has_country_flag = had_raheni_religious_turmoil 
				has_global_flag = suhan_reformation_over
			}
		} 
		is_religion_enabled = suhans_praxis
		is_force_converted = no
	}

	can_start = {
		has_global_flag = suhan_goes_sicko_mode
		has_any_disaster = no
		NOT = { religious_unity = 0.75 }
		NOT = { full_idea_group = humanist_ideas }
		any_owned_province = {
			AND = {
				has_owner_religion = no
				OR = {
					religion = high_philosophy
					religion = suhans_praxis
				}
			}
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			religious_unity = 0.75
			full_idea_group = humanist_ideas
		}
	}
	
	
	progress = {
		modifier = {
			factor = 1
			NOT = { religious_unity = 0.75 }
			religious_unity = 0.5
		} 		
		modifier = {
			factor = 2
			NOT = { religious_unity = 0.5 }
		} 
		modifier = {
			factor = 1
			has_idea_group = religious_ideas
		}
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 1
			NOT = { stability = -1 }
		}
		modifier = {
			factor = 1
			NOT = { stability = -2 }
		}
		modifier = {
			factor = 2
			has_hostile_reformation_center = yes
		}
		modifier = {
			factor = -1
			has_friendly_reformation_center = yes
		}
		modifier = { #you cannot escape
			factor = 10
			tag = R61
		}
	}
	
	
	can_end = {
		OR = {
			AND = {
				NOT = { num_of_rebel_controlled_provinces = 1 }
				OR = {
					is_force_converted = yes
					religious_unity = 0.9
					NOT = {
						any_owned_province = {
							has_owner_religion = no
							OR = {
								religion = high_philosophy
								religion = suhans_praxis
							}
						}		
					}
					full_idea_group = humanist_ideas
				}
			}
			hidden_trigger = {
				had_country_flag = { flag = active_rahen_religious_turmoil days = 9125 }
			}
		}
	}
	
	#these are hidden events which fire the apporpriate starting event
	on_start = high_philosophy_events.100	
	on_end = high_philosophy_events.103
	
	on_monthly = {
		random_events = { 
			1200 = 0
			10 = { trigger = { religion = suhans_praxis } high_philosophy_events.150 }
			10 = { trigger = { religion = suhans_praxis } high_philosophy_events.151 }
			10 = { trigger = { religion = suhans_praxis } high_philosophy_events.152 }
			10 = { trigger = { religion = suhans_praxis } high_philosophy_events.153 }
			10 = { trigger = { religion = high_philosophy } high_philosophy_events.110 }
			10 = { trigger = { religion = high_philosophy } high_philosophy_events.111 }
			10 = { trigger = { religion = high_philosophy } high_philosophy_events.112 }
			10 = { trigger = { religion = high_philosophy } high_philosophy_events.113 }
		}
	}
}
