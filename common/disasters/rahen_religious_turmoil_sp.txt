rahen_religious_turmoil = {
	potential = {
		religion_group = raheni
		has_global_flag = suhan_goes_sicko_mode
		NOT = {
			OR = {
				has_country_flag = had_religious_turmoil 
				has_global_flag = suhan_reformation_over
			}
		} 
		religion = suhans_praxis
		is_religion_enabled = suhans_praxis
		is_force_converted = no
	}

	can_start = {
		has_global_flag = suhan_goes_sicko_mode
		has_any_disaster = no
		NOT = { religious_unity = 0.75 }
		NOT = { full_idea_group = humanist_ideas }
		any_owned_province = {
			has_owner_religion = no
			religion = high_philosophy
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			religious_unity = 0.75
			full_idea_group = humanist_ideas
		}
	}
	
	
	progress = {
		modifier = {
			factor = 1
			NOT = { religious_unity = 0.75 }
			religious_unity = 0.5
		} 		
		modifier = {
			factor = 2
			NOT = { religious_unity = 0.5 }
		} 
		modifier = {
			factor = 1
			has_idea_group = religious_ideas
		}
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 1
			NOT = { stability = -1 }
		}
		modifier = {
			factor = 1
			NOT = { stability = -2 }
		}
		modifier = {
			factor = 2
			has_hostile_reformation_center = yes
		}
		modifier = {
			factor = -1
			has_friendly_reformation_center = yes
		}
	}
	
	
	can_end = {
		OR = {
			AND = {
				NOT = { num_of_rebel_controlled_provinces = 1 }
				OR = {
					is_force_converted = yes
					religious_unity = 0.9
					NOT = {
						any_owned_province = {
							has_owner_religion = no
							religion = high_philosophy
						}		
					}
					full_idea_group = humanist_ideas
				}
			}
			had_country_flag = { flag = active_religious_turmoil days = 9125 }
		}
	}
	
	on_start = rahen_religious_turmoil.151	
	on_end = rahen_religious_turmoil.152
	
	on_monthly = {
		events = {
		}
		
		random_events = { 
			1200 = 0
			10 = rahen_religious_turmoil.153
			10 = rahen_religious_turmoil.154
			10 = rahen_religious_turmoil.155
			10 = rahen_religious_turmoil.156
		
		}
	}
}
