transmutative_path_change_trade_good_trigger = {
	OR = {
		AND = {
			has_terrain = jungle
			NOT = { trade_goods = tea }
			owner = { culture = naamjyut }
		}
		AND = {
			trade_goods = cloth
			owner = { culture = khom }
		}
		AND = {
			OR = {
				has_terrain = hills
				has_terrain = highlands
				has_terrain = mountains
			}
			NOT = { trade_goods = copper }
			owner = { culture = phonan }
		}
		AND = {
			trade_goods = livestock
			owner = { culture = hinphat }
		}
		AND = {
			trade_goods = fish
			owner = { culture = binhrung }
		}
		AND = {
			trade_goods = wool
			owner = { culture = bokai }
		}
		AND = {
			trade_goods = cotton
			owner = {
				OR = {
					tag = Y86
					was_tag = Y86
				}
			}
		}
		AND = {
			OR = {
				has_terrain = drylands
				has_terrain = desert
			}
			NOT = { trade_goods = salt }
			owner = {
				OR = {
					tag = Y59
					was_tag = Y59
				}
			}
		}
		AND = {
			trade_goods = salt
			owner = {
				OR = {
					tag = Y60
					was_tag = Y60
				}
			}
		}
		AND = {
			trade_goods = grain
			owner = {
				OR = {
					tag = Y62
					was_tag = Y62
				}
			}
		}
	}
}
