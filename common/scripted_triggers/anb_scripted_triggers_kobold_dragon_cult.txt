can_unlock_kobold_cult_for_color_slot_upgrade = {
	OR = {
		has_church_aspect = $COLOUR$_dragon_1_base
		has_church_aspect = $COLOUR$_dragon_1_upgraded
	}
	NOT = {
		has_unlocked_kobold_cult_for_color_slot_upgrade = {
			COLOUR = $COLOUR$
			SLOT = $SLOT$
			UPGRADE = $UPGRADE$
		}
	}
	[[REQUIRE_BASIC]
	has_unlocked_kobold_cult_for_color_slot_upgrade = {
		COLOUR = $COLOUR$
		SLOT = $SLOT$
		UPGRADE = base
	}
	]
}
