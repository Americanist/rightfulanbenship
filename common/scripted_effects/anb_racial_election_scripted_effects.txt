racial_tolerance_define_ruler = {
	if = {
		limit = {
			has_country_flag = chose_short_lived_ruler
			OR = {
				has_country_modifier = elven_administration
				has_country_modifier = dwarven_administration
				has_country_modifier = gnomish_administration
			}
		}
		if = {
			limit = { has_country_flag = chose_harpy_ruler }
			define_ruler = {
				mil = 3
				adm = 3
				dip = 3
				culture = $culture$
				female = yes
			}
		}
		else_if = {
			limit = { republic_with_female_rulers_trigger = yes }
			define_ruler = {
				mil = 3
				adm = 3
				dip = 3
				culture = $culture$
				random_gender = yes
			}
		
		}
		else = {
			define_ruler = {
				mil = 3
				adm = 3
				dip = 3
				culture = $culture$
			}
		}
	}
	else_if = {
		limit = {
			has_country_flag = chose_short_lived_ruler
			has_country_modifier = elven_administration
		}
		if = {
			limit = { has_country_flag = chose_harpy_ruler }

	}

#the ruler is generated either from majority of a minority, if ruler is generated from minority geographical location of the target province will be used to determine culture
#Half-Elves
get_ruler_from_majority_or_minority_half_elf = {
	if = {
		limit = { has_country_modifier = half_elven_administration }
		racial_tolerance_define_ruler = { culture = ROOT }
	}
	else_if = {
		limit = {
			any_owned_province = {
				has_half_elven_majority_trigger = yes
			}
		}
		hidden_effect = {
			random_owned_province = {
				limit = { has_half_elven_majority_trigger = yes }
				save_event_target_as = half_elf_majority_province
			}
		}
		ROOT = { racial_tolerance_define_ruler = { culture = event_target:half_elf_majority_province } }
	}
	else = {
		random_owned_province = {
			limit = { has_integrated_half_elven_minority_trigger = yes }
			save_event_target_as = half_elf_minority_province
			if = {
				limit = {
					event_target:half_elf_minority_province = {
						superregion = bulwar_superregion
					}
				}
				ROOT = { racial_tolerance_define_ruler = { culture = bulwari_half_elf } }
			}
			else = {
				ROOT = { racial_tolerance_define_ruler = { culture = half_elf } }
			}
		}
	}
	
}
