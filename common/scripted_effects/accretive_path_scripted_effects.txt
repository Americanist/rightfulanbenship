		limit = {
			NOT = {
				culture_group = yan

				culture_group = harimari

				has_gov_reform = xiaken
				has_gov_reform = indep_xiaken
				has_gov_reform = wulin
				has_gov_reform = one_xia
			}
		}
		add_church_power = $amount$
	}
}
