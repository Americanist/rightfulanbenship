ogre_tolerance_giants_region = {
	if = {
		limit = {
			AND = {
				has_country_flag = ogre_tolerance_looking_for_province
				NOT = {
					has_global_flag = ogre_tolerance_giants_$region$
				}
			}
		}
		random_province = {
			limit = {
				region = $region$_region
			}
			if = {
				limit = {
					owned_by = ROOT
					fast_has_pop_of_race = {
						race = ogre
					}
				}
				province_event = {
					id = $event$
				}
				owner = {
					set_global_flag = ogre_tolerance_giants_$region$
					clr_country_flag = ogre_tolerance_looking_for_province
				}
			}
		}
	}
}
