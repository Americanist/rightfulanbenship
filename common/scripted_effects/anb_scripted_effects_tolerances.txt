ogre_tolerance_giants_region = {
	if = {
		limit = {
			NOT = {
				has_global_flag = ogre_tolerance_giants_$region$
			}
		}
		random = {
			chance = 1
			set_global_flag = ogre_tolerance_giants_$region$
			random_province = {
				limit = {
					region = $region$_region
				}
				if = {
					limit = {
						owner = {
							NOT = {
								has_country_flag = ogre_administration
							}
							has_estate = estate_adventurers
						}
					}
					province_event = {
						id = ogre_tolerance_events.12
						days = 30
						random = 15
					}
				}
				if = {
					limit = {
						owner = {
							has_country_flag = ogre_administration
						}
					}
					province_event = {
						id = ogre_flavour_events.6
						days = 30
						random = 15
					}
				}
			}
		}
	}
}
