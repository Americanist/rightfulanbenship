			duration = -1

transmutative_path_change_trade_good_effect = {
	if = {
		limit = {
			owner = { culture = naamjyut }
			has_terrain = jungle
			NOT = { trade_goods = tea }
		}
		change_trade_goods = tea
	}
	if = {
		limit = {
			owner = { culture = khom }
			trade_goods = cloth
		}
		change_trade_goods = silk
	}
	if = {
		limit = {
			owner = { culture = phonan }
			OR = {
				has_terrain = hills
				has_terrain = highlands
				has_terrain = mountains
			}
			NOT = { trade_goods = copper }
		}
		change_trade_goods = copper
	}
	if = {
		limit = {
			owner = { culture = hinphat }
			trade_goods = livestock
		}
		change_trade_goods = grain
	}
	if = {
		limit = {
			owner = { culture = binhrung }
			trade_goods = fish
		}
		change_trade_goods = gems
	}
	if = {
		limit = {
			owner = { culture = bokai }
			trade_goods = wool
		}
		change_trade_goods = grain
	}
	if = {
		limit = {
			owner = { tag = Y86 }
			trade_goods = cotton
		}
		change_trade_goods = cloth
	}
	if = {
		limit = {
			owner = {
				OR = {
					tag = Y59
					was_tag = Y59
				}
			}
			OR = {
				has_terrain = drylands
				has_terrain = desert
			}
			NOT = { trade_goods = salt }
		}
		change_trade_goods = salt
	}
	if = {
		limit = {
			owner = {
				OR = {
					tag = Y60
					was_tag = Y60
				}
			}
			trade_goods = salt
		}
		change_trade_goods = gold
	}
	if = {
		limit = {
			owner = {
				OR = {
					tag = Y62
					was_tag = Y62
				}
			}
			trade_goods = grain
		}
		change_trade_goods = livestock
	}
}
